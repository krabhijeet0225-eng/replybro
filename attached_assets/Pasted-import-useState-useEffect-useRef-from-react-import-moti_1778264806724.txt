// ── SVG Arc Meter ────────────────────────────────────────────────────────────
function ArcMeter({ value }) {
  const r = 72, cx = 90, cy = 90, strokeW = 10;
  const circumference = Math.PI * r;
  const pct = Math.min(Math.max(value / 100, 0), 1);
  const dash = pct * circumference;

  const color = value >= 70 ? "#22D3EE" : value >= 40 ? "#A78BFA" : "#EF4444";
  const label = value >= 75 ? "Very Interested 🔥" : value >= 50 ? "Kinda Interested 👀" : value >= 30 ? "Lukewarm 😐" : "Not That Into It 💀";

  return (
    <div>
      <div className="interest-meter">
        <svg width="180" height="100" viewBox="0 0 180 100">
          <path
            d={`M ${cx - r} ${cy} A ${r} ${r} 0 0 1 ${cx + r} ${cy}`}
            fill="none" stroke="rgba(255,255,255,.07)" strokeWidth={strokeW} strokeLinecap="round"
          />
          <motion.path
            d={`M ${cx - r} ${cy} A ${r} ${r} 0 0 1 ${cx + r} ${cy}`}
            fill="none" stroke={color} strokeWidth={strokeW} strokeLinecap="round"
            strokeDasharray={`${circumference} ${circumference}`}
            initial={{ strokeDashoffset: circumference }}
            animate={{ strokeDashoffset: circumference - dash }}
            transition={{ duration:1.2, ease:"easeOut" }}
          />
          <text x="90" y="82" textAnchor="middle" fill={color} fontSize="24" fontWeight="700" fontFamily="Syne,sans-serif">
            {value}%
          </text>
        </svg>
      </div>
      <p className="interest-label" style={{ color }}>{value}%</p>
      <p className="interest-sub">{label}</p>
    </div>
  );
}
